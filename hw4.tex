\begin{enumerate}
    \item \textbf{Prove $\exists N : n > N \implies s_n > a$}

    Let $\lim s_n = L, L > a$. 

    Then $\forall k \in \N \exists N \in \N \forall n \in \N : n > N \implies \lvert s_n - L \rvert < \frac{1}{k} \implies \lvert s_n - a \rvert < \frac{1}{k}$.

\end{enumerate}
